왜 1000, 2000, 3000 순서로 출력될까?

1. Callback

const echo = (time, callback) => {
    setTimeout(() => {
        console.log(time);
        callback();
    }, time);
};

const main = () => {
    echo(3000, () => {
        echo(1000, () => {
            echo(2000, () => {
                return;
            });
        });
    });
};
main();

Callback Hell

2. Promise

const echo = (time) => {
    return new Promise((resolve) => {
        setTimeout(() => {
            console.log(time);
            resolve();
        }, time);
    });
};

const main = () => {
    echo(3000)
        .then(echo.bind(null, 1000))
        .then(echo.bind(null, 2000));
};
main();

3. async / await

const echo = (time) => {
    return new Promise((resolve) => {
        setTimeout(() => {
            console.log(time);
            resolve();
        }, time);
    });
};

const main = async () => {
    await echo(3000);
    await echo(1000);
    await echo(2000);
};
main();
